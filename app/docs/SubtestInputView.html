    textInput   : "<input id='{{key}}' name='{{key}}' value='{{value}}' type='text'>"

  initialize: ( options ) ->
    @config = Tangerine.config.subtest
    @model  = options.model
    @key    = options.key
    @parent = options.parent
    @value  = options.value

  render: ->
    switch @key
      when "prototype"
        @label = "Prototype"
        @$el.html _.template(@template.label)(@) + @model.get("prototype")
      when "name"
        @label = "Name"
        @$el.html _.template(@template.label)(@) + _.template(@template.textInput)(@)
      when "enumeratorHelp"
        @label = "Enumerator Help"
        @$el.html _.template(@template.label)(@) + _.template(@template.richText)(@)
      when "studentDialog"
        @label = "Student Dialog"
        @$el.html _.template(@template.label)(@) + _.template(@template.richText)(@)
      when "items"
        @value = @model.get("items").join " "
        @label = "Grid Items"
        @$el.html _.template(@template.label)(@) + _.template(@template.textarea)(@)
      when "timer"
        @$el.html "timer unhandled"
      when "allowSkip"
        @$el.html "allow skip unhandled"
      when "gridLink"
        @$el.html "grid link unhandled"
        for subtest in @parent.subtests.models
          @$el.html "#{subtest.id}"

  getInputValue:    ->  @$el.find("input").val()
  getTextareaValue: ->  @$el.find("textarea").val()
  getCheckedValue:  ->  @$el.find("input:checked").val()
