class QuestionsEditView extends Backbone.View

  tagName : "ul"
  className : "questions_edit_view"
  
  initialize: ( options ) ->
    @views = []
    @questions = options.questions

  onClose: ->
    @closeViews()

  closeViews: ->
    for view in @views
      view.close()

  render: =>
    @closeViews()
    for question, i in @questions.models
      view = new QuestionsEditListElementView
        "question" : question
      @views.push view
      view.on "deleted", @render
      view.render()
      @$el.append view.el
